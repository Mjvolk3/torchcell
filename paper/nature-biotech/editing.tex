%% EDITING / FIGURE-SIZING version.
%% Single-column (readable for drafting) but with reduced, print-approximate
%% margins so the text block ~ the journal page width and figures size
%% realistically. NOT the submission format -- submit submission.tex (which keeps
%% the official wide-margin Nature layout). Edit content.tex (shared body), not here.
%%
%% Figure-sizing guide in this layout (A4, 182 mm text block ~ Nature's 180 mm max):
%%   \textwidth        ~ full-page-width print figure (figure*), drawn <=180 mm in draw.io
%%   ~0.49\textwidth   ~ single print column figure (~88 mm)
%% This is the WYSIWYG view: figures placed true-size (\tcfig, no scaling), so a font set
%% to N pt at <=180x170 mm in draw.io prints at exactly N pt here. To instead see the true
%% two-column print typeset, add ,iicol to the documentclass.
\documentclass[pdflatex,sn-nature,Numbered]{sn-jnl}
\input{preamble}
%% Editing-only front-matter font tweaks (drafting view; submission.tex/twocolumn.tex keep
%% the class's Nature sizes): abstract and keywords at body size (10 bp) instead of the
%% class's 9 bp / 8 bp, and affiliations at 7.5 bp so each fits on one line without wrapping.
\makeatletter
\def\abstractfont{\reset@font\fontsize{10bp}{12bp}\selectfont\leftskip=24pt\rightskip=24pt\parfillskip=0pt plus 1fil}%
\def\keywordfont{\reset@font\fontsize{10bp}{12bp}\selectfont\leftskip=24pt\rightskip=24pt plus0.5fill}%
\def\addressfont{\reset@font\fontsize{7.5bp}{9.5bp}\selectfont\titraggedcenter}%
\makeatother
%% Override the class's wide submission margins with print-approximate ones. Side margins
%% give a 182 mm text block (A4 210 - 2*14) so a true 180 mm figure fits flush at 1:1.
%% twoside=false: the class is \LoadClass[twoside]{article}, so geometry treats left/right
%% as inner/outer and swaps them per page (binding margins). For the drafting view we want
%% symmetric, non-alternating margins on every page, so force one-sided margins here.
\geometry{a4paper,left=14mm,right=14mm,top=15mm,bottom=22mm,twoside=false,bindingoffset=0mm,marginparwidth=0pt,marginparsep=0pt}
% \cb = word-budget tag (approx words; ~6 chars/word), shown only in this editing build
% (invisible in submission.tex / twocolumn.tex, which define \cb as a no-op).
\newcommand{\cb}[1]{\ {\normalfont\small\color{gray}(#1)}}
% \wc{id} = editing-only ACTUAL word count next to a heading, in brackets. The wc@<id>
% macros are generated by `wordcount.py --tex wordcount.tex` (input below, before the body)
% and expand to a colored "words/budget"; \ifcsname guards a missing/stale file.
% submission.tex / twocolumn.tex define \wc as a no-op, so the shared headings still compile.
\newcommand{\wc}[1]{\ifcsname wc@#1\endcsname\,{\normalfont\footnotesize[\csname wc@#1\endcsname]}\fi}
% Drafting view only: block paragraphs -- no first-line indent, vertical gap between paras.
% submission.tex / twocolumn.tex keep the class's indented Nature default.
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.5\baselineskip plus 0.1\baselineskip}
\begin{document}
% Editing-only word counts. Input BEFORE the body so the inline \wc{id} macros exist when
% the headings typeset; this also defines \wcsummarytable (rendered at the end). The Makefile
% regenerates wordcount.tex from the sources before this view compiles; \IfFileExists guards
% a fresh checkout that hasn't generated it yet.
\IfFileExists{wordcount.tex}{\input{wordcount}}{}
\input{content}
\ifcsname wcsummarytable\endcsname\wcsummarytable\fi
\end{document}
